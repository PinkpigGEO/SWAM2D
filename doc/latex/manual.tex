\begin{eqnarray}

\noindent Fourth-order forward ($D^{+}$) and backward ($D^{-}$) operators:
\begin{eqnarray}
  D^{+}=c_{1}[f(i+1)-f(i)]+c_{2}[f(i+2)-f(i-1)] \nonumber \\

\end{eqnarray}
